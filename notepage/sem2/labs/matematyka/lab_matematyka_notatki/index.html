Notatka - wektory 

        # Tak tworzymy wektor
        a = [1 2 3 4 5]

        # Pobranie długości wektora
        lenofx = length(x)

        # Pomnóż każdy element przez 4
        a * 4

        # Spotęguj każdy elment do potęgi 2 (kropka jest potrzebna, żeby Octave nie pomyślał, że chodzi o potęgowanie macierzy)
        a .^ 2 

        # Indeksowanie zaczyna się od 1 (nie od 0 jak w większości j. programowania)
        a(3)

        # Wyświetl 1 i 4 indeks
        a([1 4]) 

        # Podmieni 1 i 4 indeks na inne wartości
        a([1 4]) = [6 -3] 

        # Tworzenie wektorów metodą start:step:stop
        b = 1:2:10

        # Pominięcie środkowej wartości powoduje step=1 (metoda start:stop)
        b = 1:10 #

        # Tworzy wektor z 3 równo rozmieszczonymi na osi liczbowej liczbami, z których pierwsza to 1, a druga to 5
        c = linspace(1,5,3) 

        # Sumuje elementy wektora
        d = sum(linspace(1, 3, 100))

        # Selekcja warunkowa, wybiera z wektora "a" elementy więsze od zera
        a(a>0)
Notatka - macierze 

        # Przykładowa macierz kwadratowa (średnik kończy wiersz)
        x = [1 2 3; 1 2 3; 1 2 3]

        # Odwoływanie się do wiesza i kolumny
        x(1, 2)  # 1 wiersz, 2 kolumna

        # Pobierz wszystkie wiersze i 1 kolumnę
        x(:, 1)

        # Pobierz wszystkie kolumny i drugi wiersz
        x(2, :)

        # Pobierz 1 i 2 wiersz z 3 i 4 kolumną
        x(1:2, 3:4)

        # Pobierz pierwszą kolumnę dla każdego wiersza, w którym 3 kolumna równa się 12
        t(t(:,3) == 12, 1)
